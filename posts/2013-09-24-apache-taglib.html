태그를 사용하기 위한 선언

<%@ taglib prefix="c" uri="http://java.sun.com/jsp/jstl/core"%>
<%@ taglib prefix="fn" uri="http://java.sun.com/jsp/jstl/functions" %>
<%@ taglib prefix="fmt" uri="http://java.sun.com/jsp/jstl/fmt"%>

param

test.jsp?name=1234 일 경우

${param.변수} 로 획득할 수 있다.
${param.name}으로 알수 있겠다.

core

set

<c:set var="string" value="값"/>
<c:set var="string" value="${변수.프로퍼티}"/>

out

<c:out value="${fn:substring(string,0,50)}"  default="초기값"/>
<c:out value="안녕하세요" />

test

<c:if test="${변수가 존재하면}">
..처리
</c:if>

<c:if test="${변수 < 숫자 }">
..처리
</c:if>

<c:if test="${변수 < true}">
..처리
</c:if>

choose

<c:choose>
  <c:when test="조건">
 ..
  </c:when>
  <c:when test="조건">
 ..
  </c:when>
  <c:otherwise>
   그외 조건처리
  </c:otherwise>
</c:choose>

exception

<c:catch var="exception">
 ...처리
</c:catch>

url

<c:url value='/hello.jsp'/>"

function

length

<c:set var="length" value="${fn:length(변수)}"/>

substring

<c:out value="${fn:substring(string,0, fn:length(string))}" />

format

formatDate

<fmt:formatDate value="${변수}" pattern="패턴" />

sql

<%@ page pageEncoding="utf-8" %>
<%@ taglib prefix="sql" uri="http://java.sun.com/jsp/jstl/sql" %>
<%@ taglib prefix="c" uri="http://java.sun.com/jsp/jstl/core" %>

<sql:setDataSource dataSource="jdbc/wodb"/>

<sql:query var="rs"  sql="select 1 as id from dual union all select 10 as id from dual"/>
 or
<sql:query var="rs">
  select 1 as id from dual union all select 10 as id from dual"
</sql:query>

<table border="1">
<c:forEach var="rec" items="${rs.rows}" >
　　<tr> 
　　　　<td><c:out value="${rec.id}" /></td>
　　</tr>
</c:forEach>
</table>
