컴포넌트는 다른 컴포넌트를 포함할 수 있다

var Row = React.createClass({

	render: function() {
		return <tr><td>{this.props.name}</td></tr>;
	}
});

var Table = React.createClass({

	render: function() {

	var data = [];
	data.push(<Row name="hello1"/>);
	data.push(<Row name="hello2"/>);
	data.push(<Row name="hello3"/>);

		return <table className="table"><thead>
		  <tr>
		    <th>Attribute</th>
		    <th>Value</th>
		  </tr>
		  </thead>
		  <tbody>{data}</tbody>
		  </table>
	}
});

React.render(<Table />, document.getElementById('hello'));

State Component

입력폼 만들기

var input = React.createClass({}); //input는 html이므로 작동이 안된다. 아래처럼 수정할 것

var Input = React.createClass({
  render: function() {
    return <input type="text" value="Hello there" />;
});

초기값을 설정

var Input = React.createClass({
	getInitialState: function() {				
		return {abc: "Hello There!"};
	},
	
	render: function() {
		return <Input type="text" value={this.state.abc} />;
	}
})

this.state는 컴포넌트의 상태값을 가지고 있다
동적으로 컴포넌트의 상태값을 바뀌기 위해서
this.state에 변화를 주면 된다.

변화를 주기 위해서

this.setState( 데이터 )

Event 핸들러

text 입력창에 변화가 발생할 때마다 초기값으로 돌리는 예이다
결과는 입력값 수정이 불가능하게 된다

var Input = React.createClass({
	getInitialState: function() {
		return {abc: "Hello There!"};
	},
	handleEvent: function(e) {
		this.setState({abc: this.state.abc});
	},
	render: function() {
		return <input type="text" value={this.state.abc} onChange={this.handleEvent}/>;
	}
})
