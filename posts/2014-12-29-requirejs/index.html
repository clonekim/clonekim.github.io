RequireJS 예제

RequireJS

log라는 모듈을 작성한다

log.js

define('log', function() {
	
	return function(str) {
		console.log("--> ", str);
	}
});

<!DOCTYPE html>
<html lang="ko">
<head>
   
    <script src="requirejs/2.1.15/require.js"></script>
    <script src="js/log.js"></script>
</head>
<body>

<script>
 require(['log'], function(log) {
 	log("Hello !");
 });
</script>
</body>
</html>

프로토타입으로 작성한 boost.js 

define('boost', function() {
	
	function Boost() {

	}

	Boost.prototype.version = function() {
		return "1.0";
	};

	return Boost;

});

<!DOCTYPE html>
<html lang="ko">
<head>
  <script src="requirejs/2.1.15/require.js"></script>
  <script src="js/boost.js"></script>
  <script>
   require(['boost'], function(Boost) {
	var boost = new Boost();
	alert(boost.version());
   });
</script>

IIFE(mmediately-Invoked Function Expression) 를 적용하면
new 생성자 없이 사용가능하다

define('boost', function() {
	
	var boost = (function(){
		var version = "1.0";

		return {
			version: function() {
				return version; 
			}
		}
	})();

	return boost;
});
