AngularJS 다국어처리

IndexController.js

module.exports = ['$scope', '$cookies', 'TranslationService', function($scope, $cookies, TranslationService) {
    TranslationService.getLanguage($scope, $cookies);

    $scope.multiLng = function( lang ) {
        $cookies.put("lang", lang);
        TranslationService.getLanguage($scope, $cookies);
    }
}];

TranslationService.js

module.exports = ['$http', function($http) {

	this.getLanguage = function($scope, $cookies, cbk) {

		var lang = $cookies.get('lang') || 'ko';
		
		$http({
			method: 'GET',
			url: '/js/locale_' + lang +'.json'
		}).success(function(response) {
		    $scope.translation = response;		    
		    $scope.lang = lang;                    
		});
	};
}];
