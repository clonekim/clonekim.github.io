Net:HTTP

GET/PUT

require 'net/http'
http = Net::HTTP.new('www.data.com')
response = http.request_put('/?access_token=123', jsonbody)

url = URI.parse('http://localhost:3000/someservice/')
request = Net::HTTP::Post.new(url.path)
request.body = "<?xml version='1.0' encoding='UTF-8'?><somedata><name>Test Name 1</name><description>Some data for Unit testing</description></somedata>"
response = Net::HTTP.start(url.host, url.port) {|http| http.request(request)}

#Note this test PASSES!
assert_equal '201 Created', response.get_fields('Status')[0]

require 'net/http'
require 'uri'

type = 'record'
activity = 'query-data'

xml_req =<<EOF
<? xml version='1.0' ?>
<DATASET>
<SITE_ID>123456789</SITE_ID>
<MLID>987654321</MLID>
<DATA type='email'>user@somedomain.com</DATA>
</DATASET>
EOF


req = "type=#{type}&activity=#{activity}&input="
req = req + URI.escape(xml_req)

res = Net::HTTP.start('www.uptilt.com') { |http|
  http.post("/API/mailing_list.html", req)
}
puts res.body

IO

popen

IO.popen명령어를 사용해서 IO오브젝트를 얻는다
get을 호출하면 출력결과를 볼 수 있다

IO.popen("ping localhost") do |io|
  while io.gets
    print $_
  end
end
