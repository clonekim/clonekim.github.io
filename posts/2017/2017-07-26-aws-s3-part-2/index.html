AWS S3 Part 2

S3 Condition

Referer를 참조해서 리소스제한

{
    "Version": "2012-10-17",
    "Id": "Policy1463129932149",
    "Statement": [
        {
            "Sid": "Stmt1463129927965",
            "Effect": "Allow",
            "Principal": {
                "AWS": "*"
            },
            "Action": "s3:GetObject",
            "Resource": "arn:aws:s3:::your-bucket/*",
            "Condition": {
                "StringLike": {
                    "aws:Referer": [
                        "http://yoursite/*",
                        "http://*.yoursite/*"
                    ]
                }
            }
        }
    ]
}
