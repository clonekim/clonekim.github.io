패키지 설치

npm install --global gulp
npm install --save-dev gulp
npm install --save-dev browserify
npm install --save-dev vinyl-source-stream
npm install --save handlebars
npm install --save hbsfy

gulpfile.js

var gulp = require('gulp'),
	browserify = require('browserify'),
	source = require('vinyl-source-stream'),
	hbsfy = require('hbsfy');

gulp.task('build', function() {
	var b = browserify('./js/app.js', {debug:true});
	return b.transform(hbsfy)
		 .bundle()
		 .pipe(source('app.js'))
		 .pipe(gulp.dest('/home/alpha/public/js'));
});


gulp.task('watch', function() {
	gulp.watch('./js/**/*.js', ['build']);
	gulp.watch('./templates/*.hbs', ['build']);
});

app.js

var Backbone = require('backbone');
var $ = require('jquery');
var HomeView = require('./views/home');

var Router = Backbone.Router.extend({

	
		routes: {
			'': 'start'
		},
		
		start: function() {
			console.log('start!');
			$(".container").html(new HomeView().render().el);
		}
		
});

var router = new Router();
Backbone.history.start();

views/home.js

var Backbone = require('backbone');
var $ = require('jquery');
Backbone.$ = $;


var HomeView = Backbone.View.extend({
	initialize: function() {
		this.template = require('../../templates/home.hbs');
	},
	
	render: function() {
		this.$el.html(this.template({name: 'Hello There!'}));
		console.log("home render");
		return this;
	}
});

module.exports = HomeView;
