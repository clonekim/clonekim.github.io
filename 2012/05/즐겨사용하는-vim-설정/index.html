즐겨사용하는 VIM 설정

~./vimrc 또는 /etc/vim/vimrc.local

set autoindent
set smartindent
set shiftwidth=3
set tabstop=3
set title
set nu
set background=dark
set nobackup
set fileencoding=utf-8
set nowrap
set showmatch
set ruler
syntax on
#vi
