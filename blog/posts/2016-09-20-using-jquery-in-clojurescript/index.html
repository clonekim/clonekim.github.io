clojurescript에서 jquery사용

view.clj 

(defn view-layout [& content]
  (html
      [:head
           [:meta {:http-equiv "Content-type"
                        :content "text/html; charset=utf-8"}]
           ["title": "Project-Name"]]
      [:body content]))

(defn view-content []
  (view-layout
       [:h2 "Project-Name"]
       [:p {:id "clickhere"} "Get yourself a nice alert by clicking here."]
       [:script {:src "/js/jquery-1.10.2.min.js"}]
       [:script {:src "/js/cljs.js"}]))

hello.cljs

(:require [jayq.core :as jq])
(:use [jayq.core :only [$]])

;;
;; $("#btn").click(function() {alert("Hi!"); });
;;

(jq/bind ($ "#btn") :click (fn [] (js/alert "Hi!")))

lein cljsbuild once
lein run
