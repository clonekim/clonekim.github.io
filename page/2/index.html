(ac-source-css-property)) ("html" . (ac-source-words-in-buffer ac-source-abbrev))))) (add-hook 'web-mode-hook 'my-web-mode-hook) (add-hook 'web-mode-before-auto-complete-hooks '(lambda () (let ((web-mode-cur-language (web-mode-language-at-pos))) (if (string= web-mode-cur-language "html") (setq emmet-use-css-transform t) (setq emmet-use-css-transform nil)) (if (string= web-mode-cur-language "css") (setq emmet-use-css-transform t) (setq emmet-use-css-transform nil)))))
